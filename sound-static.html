  #wrap{display:flex;flex-direction:column;gap:12px;align-items:center;justify-content:center;height:100%}

  canvas{image-rendering: pixelated;border:1px solid #222}

  button{font-size:16px;padding:10px 16px;border-radius:10px;border:0;background:#1e90ff;color:#fff}

</style>

<div id="wrap">

  <canvas id="c" width="256" height="192"></canvas>

  <button id="start">Start</button>

</div>

<script>

(() => {

  const canvas = document.getElementById('c');

  const ctx = canvas.getContext('2d');

  const btn = document.getElementById('start');

  // schaal canvas mooi naar schermbreedte

  const fit = () => {

    const scale = Math.min(window.innerWidth * 0.95 / canvas.width, window.innerHeight * 0.8 / canvas.height);

    canvas.style.width = (canvas.width * scale) + 'px';

    canvas.style.height = (canvas.height * scale) + 'px';

  };

  window.addEventListener('resize', fit); fit();

  let audioCtx, analyser, data, running = false;

  async function start() {

    if (running) return;

    running = true;

    // AudioContext MOET via een echte gebruikers-tap worden gestart (iOS/Chrome-policy)

    audioCtx = new (window.AudioContext || window.webkitAudioContext)();

    await audioCtx.resume(); // binnen UI-event

    const stream = await navigator.mediaDevices.getUserMedia({ audio: { echoCancellation:false, noiseSuppression:false, autoGainControl:false }});

    const src = audioCtx.createMediaStreamSource(stream);

    analyser = audioCtx.createAnalyser();

    analyser.fftSize = 2048;

    src.connect(analyser);

    data = new Uint8Array(analyser.fftSize);

    btn.textContent = 'Bezig… (tik om pauze)';

    loop();

  }

  function stop() {

    running = false;

    btn.textContent = 'Start';

  }

  btn.addEventListener('click', () => running ? stop() : start());

  function micLevel() {

    analyser.getByteTimeDomainData(data); // tijdsdomein-samples ophalen

    // RMS-niveau 0..1

    let sum = 0;

    for (let i=0;i<data.length;i++){

      const v = (data[i]-128)/128;

      sum += v*v;

    }

    return Math.sqrt(sum / data.length);

  }

  function loop() {

    if (!running) return;

    const level = micLevel(); // 0..1

    // kies willekeurige pixelpositie

    const x = (Math.random() * canvas.width) | 0;

    const y = (Math.random() * canvas.height) | 0;

    // drempel dynamisch: hoe harder het geluid, hoe meer witte pixels

    const threshold = 0.06 + Math.random()*0.14; // ongeveer rustig→luid

    if (level >= threshold) {

      ctx.fillStyle = '#fff';

      ctx.fillRect(x, y, 1, 1); // 1x1 pixel tekenen

    } else {

      ctx.fillStyle = '#000';

      ctx.fillRect(x, y, 1, 1);

      // af en toe schoonvegen voor “flikker” gevoel

      if (Math.random() < 0.001) ctx.fillRect(0, 0, canvas.width, canvas.height);

    }

    requestAnimationFrame(loop);

  }

})();

</script>
